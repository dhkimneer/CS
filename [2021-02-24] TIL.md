## Q56. Implement Trie (Prefix Tree)  (Leetcode Q208)

<br>

트라이의 insert, search, startsWith 메소드를 구현하라. (starts_with으로 밑에 바꿔서 씀)

<br>

```python
Trie trie = new Trie();

trie.insert("apple");
trie.search("apple"); // returns true
trie.search("app");  // returns false
trie.starts_with("app");  // returns true
trie.insert("app");
trie.search("app");  // returns true
```

#### [풀이1. 딕셔너리를 이용해 간결한 트라이 구현]

<br>

```python
import collections


# 트라이의 노드
class TrieNode:
    def __init__(self):
        self.word = False
        self.children = collections.defaultdict(TrieNode)  # default value가 TrieNode인 dictionary
        
        
class Trie:
    def __init__(self):
        self.root = TrieNode()
        
    # 단어 삽입
    def insert(self, word: str) -> None:
        node = self.root  # root node는 특정 문자를 의미하지 않고, 자식 노드만 가지고 있음
        for char in word:
            node = node.children[char]
            # key는 하나의 알파벳, value는 key에 해당하는 자식노드 // 확인
        node.word = True
        
    # 단어 존재 여부 판별
    def search(self, word: str) -> bool:
        node = self.root
        for char in word:
            if char not in node.children:
                return False
            node = node.children[char]
            
        return node.word
    
    # 문자열로 시작 단어 존재 여부 판별
    def starts_with(self, prefix: str) -> bool:
        node = self.root
        for char in prefix:
            if char not in node.children:
                return False
            node = node.children[char]
            
        return True
```

## Q57. Palindrome Pairs (Leetcode Q336)

<br>

단어 리스트에서 words[i] + words[j]가 팰린드롬이 되는 모든 인덱스 조합 (i, j)를 구하라.

예제1)

입력: ["abcd", "dcba", "lls", "s", "sssll"]

출력: [[0, 1], [1, 0], [3, 2], [2, 4]]

예제2)

입력: ["bat", "tab", "cat"]

출력: [[0, 1], [1, 0]]

<br>

#### [풀이1. 팰린드롬을 브루트 포스로 계산] : Timeout

<br>

```python
from typing import List


class Solution:
    def palindrome_pairs(self, words: List[str]) -> List[List[int]]:
        def is_palindrome(word):
            return word == word[::-1]
        
        output = []
        for i, word1 in enumerate(words):
            for j, word2 in enumerate(words):
                if i == j:  # index 같을 경우 다음 순번으로 넘어가
                    continue
                if is_palindrome(word1 + word2):
                    output.append([i, j])
        return output
```

#### [풀이2. 트라이 구현]

<br>

```python
import collections
from typing import List


# 트라이 저장할 노드
class TrieNode:
    def __init__(self):
        self.children = collections.defaultdict(TrieNode)
        self.word_id = -1
        self.palindrome_word_ids = []
        
       
class Trie:
    def __init__(self):
        self.root = TrieNode()
        
    @staticmethod  # class instance에 접근하는 것 제한, 독립적인 함수 // decorator
    def is_palindrome(word: str) -> bool:  # 그래서 self 없음
        return word[::] == word[::-1]
    
    # 단어 삽입
    def insert(self, index, word) -> None:
        node = self.root
        for i, char in enumerate(reversed(word)):
            if self.is_palindrome(word[0:len(word) - i]):
                node.palindrome_word_ids.append(index)
            node = node.children[char]
        node.word_id = index
        
    def search(self, index, word) -> List[List[int]]:
        result = []
        node = self.root
        
        while word:
            # 판별 로직 3)
            if node.word_id >= 0:
                if self.is_palindrome(word):
                    result.append([index, node.word_id])
            if not word[0] in node.children:
                return result
            node = node.children[word[0]]
            word = word[1:]
            
        # 판별 로직 1)
        if node.word_id >= 0 and node.word_id != index:
            result.append([index, node.word_id])
            
        # 판별 로직 2)
        for palindrome_word_id in node.palindrome_word_ids:
            result.append([index, palindrome_word_id])
            
        return result
    

class Solution:
    def palindrome_pairs(self, words: List[str]) -> List[List[int]]:
        trie = Trie()
        
        for i, word in enumerate(words):
            trie.insert(i, word)
            
        results = []
        for i, word in enumerate(words):
            results.extend(trie.search(i, word))
            
        return results
```
