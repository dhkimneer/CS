```python
from typing import List


class Solution:
    def diff_ways_to_compute(self, input: str) -> List[int]:
        def compute(left, right, op):
            results = []
            for l in left:
                for r in right:
                    results.append(eval(str(1) + op + str(r)))
            return results
        
        if input.isdigit():
            return [int(input)]
        
        results = []
        for index, value in enumerate(input):
            if value in "-+*":
                left = self.diff_ways_to_compute(input[:index])
                right = self.diff_ways_to_compute(input[index + 1:])
                
                results.extend(compute(left, right, value))
        return results
```

__Ch23. 다이나믹 프로그래밍__

<br>

- 문제를 각각의 작은 문제로 나누어 해결한 결과를 저장해뒀다가 나중에 큰 문제의 결과와 합하여 풀이하는 알고리즘

```python
[패러다임]

최적 부분 구조 + 중복된 하위 문제들

[방법론]

1. 상향식(Bottom-Up) (= Tabulation)
# 이 방식만을 다이나믹 프로그래밍으로 지칭하기도 함

더 작은 하위 문제부터 살펴본 다음, 작은 문제의 정답을 이용해 큰 문제의 정답을 풀어나가는 방식

def fib(n):
    dp[0] = 0
    dp[1] = 1
    
    for i in range(2, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2]
    return dp[n]

2. 하향식(Top-Down) (= Memoization)

하위 문제에 대한 정답을 계산했는지 확인해가며 문제를 자연스레 풀어나가는 방식

def fib(n):
    if n <= 1:
        return n
    
    if dp[n]:
        return dp[n]
    dp[n] = fib(n - 1) + fib(n - 2)
    return dp[n]
```

```python
import collections


class Solution:
    dp = collections.defaultdict(int)
    
    def fib(self, N: int) -> int:
        if N <= 1:
            return N
        
        if self.dp[N]:  # 저장해놓았던 값을 바로 리턴
            return self.dp[N]
        self.dp[N] = self.fib(N - 1) + self.fib(N - 2)
        return self.dp[N]
```

#### [풀이3. Tabulation (Bottom-Up)]

<br>

```python
import collections


class Solution:
    dp = collections.defaultdict(int)
    
    def fib(self, N: int) -> int:
        self.dp[1] = 1
        
        for i in range(2, N + 1):
            self.dp[i] = self.dp[i - 1] + self.dp[i - 2]
        return self.dp[N]
```

```python
    (2, 1),
    (10, 4),
    (1, 1),
    (2, 2),
]


def zero_one_knapsack(cargo):
    capacity = 15
    pack = []
    
    for i in range(len(cargo) + 1):
        pack.append([])
        for c in range(capacity + 1):
            if i == 0 or c == 0:
                pack[i].append(0)
            elif cargo[i - 1][1] <= c:
                pack[i].append(
                	max(
                    	cargo[i - 1][0] + pack[i - 1][c - cargo[i - 1][1]],
                        pack[i - 1][c]
                    ))
            else:
                pack[i].append(pack[i - 1][c])
    return pack[-1][-1]


r = zero_one_knapsack(cargo)
print(r)
```
